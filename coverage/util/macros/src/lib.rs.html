66/// Polls all passed [`Future`](std::future::Future)s or
67/// [`Stream`](std::stream::Stream)s until one is ready and returns
68/// it. Prioritizes by order.
69///
70/// # Raw-mode
71/// If the `complete` item is unspecified, [`Future`](std::future::Future)s and
72/// [`Stream`](std::stream::Stream)s will be pulled beyond exhaustion. If a high
73/// priority item is exhausted and [`select!`](crate::select!) is being used in
74/// a loop, it will continuously return the same result, which might be
75/// unintended.
76///
77/// # Fused-mode
78/// If the `complete` item is specified, all polled items have to implement
79/// `FusedFuture` or `FusedStream`, for [`Stream`](std::stream::Stream)s this
80/// will unwrap the [`Option`]. The `complete` item will be executed once all
81/// items have been exhausted. Items will not be pulled beyond exhaustion.
82///
83/// # Examples
84/// Raw-mode:
85/// ```
86/// # futures_executor::block_on(async {
87/// use futures_util::stream;
88/// use allochronic_util::select;
89///
90/// let mut stream1 = stream::iter(1..=3);
91/// let mut stream2 = stream::iter(4..=6);
92/// let mut stream3 = stream::iter(7..=9);
93///
94/// let mut counter = 0;
95///
96/// while let Some(result) = select![
97/// result: &mut stream1 => result,
98/// result: &mut stream2 => result,
99/// result: &mut stream3 => result,
100/// ] {
101/// counter += 1;
102/// assert_eq!(counter, result);
103/// }
104///
105/// // this loop will only reach the number of three, because it will abort after the first stream
106/// // is exhausted
107/// assert_eq!(counter, 3);
108/// # });
109/// ```
110/// Fused-mode:
111/// ```
112/// # futures_executor::block_on(async {
113/// use futures_util::stream::{self, StreamExt};
114/// use allochronic_util::select;
115///
116/// let mut stream1 = stream::iter(1..=3).fuse();
117/// let mut stream2 = stream::iter(4..=6).fuse();
118/// let mut stream3 = stream::iter(7..=9).fuse();
119///
120/// let mut counter = 0;
121///
122/// while let Some(result) = select![
123/// result: &mut stream1 => Some(result),
124/// result: &mut stream2 => Some(result),
125/// result: &mut stream3 => Some(result),
126/// complete => None,
127/// ] {
128/// counter += 1;
129/// assert_eq!(counter, result);
130/// }
131///
132/// assert_eq!(counter, 9);
133/// # });
134/// ```
135#[proc_macro]
13620pub fn select(item: TokenStream) -> TokenStream {
13720 select::select(item)
