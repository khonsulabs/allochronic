12/// [`Poll`](std::future::Future::poll)s a [`FusedFuture`] a single time,
13/// returning [`Poll<Option<T>>`]. Used in combination with
14/// [`PollFutures`](super::PollFutures::__into_poll_fused).

16/// Will always return [`None`] if [`is_terminated`](FusedFuture::is_terminated)
17/// returns `true`.
18///
19/// # Examples
20/// ```
21/// # futures_executor::block_on(async {
22/// use std::task::Poll;
23///
24/// use allochronic_util::select::fused;
25/// use futures_util::future::{self, FutureExt};
26///
27/// // `ready` already produces a fused `Future`, `fuse()` here is just to showcase
28/// let mut future = future::ready(1).fuse();
